SELECT RIGHT(col_name, 2) FROM my_contacts;

//SUBSTRING_INDEX(col,' ',n)函数
SELECT SUBSTRING_INDEX (location, ',', 1)    //找到第1个","，并截取前面的部分
    FROM my_contacts;
